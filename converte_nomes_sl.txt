UNICODE_STRING prefixNT = RTL_CONSTANT_STRING(L"\\??\\");

VOID winRoot_setup() {
    RtlInitUnicodeString(&winRoot, SharedUserData->NtSystemRoot);

// Alocaçăo de memória compatível com Windows 7 até o 11
// NonPagedPool e POOL_FLAG_NON_PAGED: Memória que nunca sai da RAM (essencial para drivers)
static PVOID Malloc(SIZE_T Size, ULONG Tag) {
#if (NTDDI_VERSION >= NTDDI_WIN8)
    if (g_Debug) DbgPrint("[SENTINELA] Utilizando ExAllocatePool2 para alocar RAM.\n");
    return ExAllocatePool2(POOL_FLAG_NON_PAGED, Size, Tag);
#else
    if (g_Debug) DbgPrint("[SENTINELA] Utilizando ExAllocatePoolWithTag para alocar RAM.\n");
    return ExAllocatePoolWithTag(NonPagedPool, Size, Tag);
#endif
}

VOID copia_unicode_data(PUNICODE_STRING Entrada, PUNICODE_STRING Saida) {
    // Copia o nome original e ja era
    Saida->Buffer = Entrada->Buffer;
    Saida->Length = Entrada->Length;
    Saida->MaximumLength = Entrada->MaximumLength;
}

USHORT calcula_tamanho_sufixo_device_name(PUNICODE_STRING deviceName) {
    // inicializando pro tamanho maximo
    USHORT retval = deviceName->MaximumLength;
    for (USHORT i = 0; i < deviceName->MaximumLength / sizeof(WCHAR); i++) {
        if (deviceName->Buffer[i] == L'\\') {
            retval = (USHORT)(i * sizeof(WCHAR));
            break;
        }
    }
    return retval;
}

USHORT tamanho_deviceName(PUNICODE_STRING FileName) {
    UNICODE_STRING sufixDevName;
    sufixDevName.Buffer = (PWCH)((PCHAR)FileName->Buffer + prefixDevice.Length);
    sufixDevName.MaximumLength = FileName->Length - prefixDevice.Length;
    //primeiro percorre FileName por "\" apos o tamanho de prefixDevice para obter o nome do dispositivo (ex: \Device\HarddiskVolumeX)
    return prefixDevice.Length + calcula_tamanho_sufixo_device_name(&sufixDevName);
}

BOOLEAN caso15(PUNICODE_STRING FileName, PUNICODE_STRING FileNameOk, USHORT tamanhoDevName_cache) {
    UNICODE_STRING completeDeviceName = { 0 };
    UNICODE_STRING sufixo = { 0 };
    UNICODE_STRING dosName = { 0 };
    UNICODE_STRING tmpDosName = { 0 };

    BOOLEAN all_ok = FALSE;

    NTSTATUS status_objptr = STATUS_ACCESS_DENIED;
    NTSTATUS status_dosname = STATUS_ACCESS_DENIED;

    PFILE_OBJECT fileObject = NULL;
    PDEVICE_OBJECT deviceObject;

    // checa a condicao do caso 1 classico
    // se for igual a g_DevicePrefix entao dosname ja eh conhecido
    if (RtlPrefixUnicodeString(&g_DevicePrefix, FileName, TRUE)) {
        dosName.Buffer = g_winSystem32Path.Buffer; // Supondo que g_winSystem32Path tenha "C:\..."
        dosName.Length = 4; // "C:" 4 bytes em wchar
        dosName.MaximumLength = 4;

        sufixo.Buffer = (PWSTR)((PCHAR)FileName->Buffer + g_DevicePrefix.Length);
        sufixo.Length = FileName->Length - g_DevicePrefix.Length;
        sufixo.MaximumLength = sufixo.Length;
        
        all_ok = TRUE;
    }
    else {
        
        //montando a string completa do completeDeviceName reutilizando o buffer de filename
        completeDeviceName.Buffer = FileName->Buffer;
        completeDeviceName.Length = tamanhoDevName_cache;
        completeDeviceName.MaximumLength = tamanhoDevName_cache;

        // Calcula o sufixo do caminho original (o que vem após o nome do dispositivo) para anexar depois
        // reutilizando o buffer de FileName
        sufixo.Buffer = (PWCH)((PCHAR)FileName->Buffer + tamanhoDevName_cache);
        sufixo.Length = FileName->Length - tamanhoDevName_cache;
        sufixo.MaximumLength = sufixo.Length;

        // obtendo o objeto de dispositivo do completedevicename encontrado
        // derreferenciar se tiver sucesso
        status_objptr = IoGetDeviceObjectPointer(
            &completeDeviceName,
            FILE_READ_ATTRIBUTES,
            &fileObject,
            &deviceObject
        );

        if (NT_SUCCESS(status_objptr)) {
            // Agora vocę tem o deviceObject para usar em IoVolumeDeviceToDosName
            // e assim obter o dosname do completedevicename encontrado
            // desalocar dosName se tiver sucesso
            status_dosname = IoVolumeDeviceToDosName(deviceObject, &tmpDosName);
        }
        else {
            DbgPrint("[SENTINELA] converte_nome_arquivo: - Erro - caso1.5: falha ao chamar IoGetDeviceObjectPointer: 0x%08X\n", status_objptr);
        }

        if (NT_SUCCESS(status_dosname)) {
            // checa se o tamanhao de tmpDosName é 4
            if (tmpDosName.Length >= 4) {
                dosName.Buffer = tmpDosName.Buffer;
                dosName.Length = 4;
                dosName.MaximumLength = 4;
                all_ok = TRUE;
            }
            else {
                DbgPrint("[SENTINELA] converte_nome_arquivo: - Erro - caso1.5: caminho retornado por IoVolumeDeviceToDosName tem menos caracteres do que o esperado (4): %wZ.\n", tmpDosName);
            }
        }
        else {
            DbgPrint("[SENTINELA] converte_nome_arquivo: - Erro - caso1.5: falha ao chamar IoVolumeDeviceToDosName: 0x%08X\n", status_dosname);
        }

    }

    
    if (all_ok && FileNameOk->MaximumLength >= (dosName.Length + sufixo.Length)) {
        // Copia a letra da unidade (ex: C:)
        RtlCopyMemory(FileNameOk->Buffer, dosName.Buffer, dosName.Length);
        FileNameOk->Length = dosName.Length;
        // Anexa o sufixo original
        RtlCopyMemory((PCHAR)FileNameOk->Buffer + FileNameOk->Length, sufixo.Buffer, sufixo.Length);
        FileNameOk->Length += sufixo.Length;
    }
    else {
        all_ok = FALSE;
    }
    

    // DENTRO DO caso15:
    /*
    if (all_ok) {
        // 1. Limpa o buffer de destino com segurança
        NTSTATUS status = RtlUnicodeStringCopy(FileNameOk, &dosName);

        if (NT_SUCCESS(status)) {
            status = RtlUnicodeStringCat(FileNameOk, &sufixo);
            if (!NT_SUCCESS(status)) {
                all_ok = FALSE;
                DbgPrint("[SENTINELA] converte_nome_arquivo: - Erro - caso1.5: Năo foi possível concatenar sufixo: %wZ\n", sufixo);
            }
        }
        else {
            all_ok = FALSE;
            DbgPrint("[SENTINELA] converte_nome_arquivo: - Erro - caso1.5: Năo foi possível concatenar dosName: %wZ\n", dosName);
        }
    }*/

    // LIMPEZA CENTRALIZADA (Único lugar para evitar BSOD 18)
    if (NT_SUCCESS(status_objptr) && fileObject != NULL) {
        ObDereferenceObject(fileObject);
    }

    // tem que desalocar antes de retornar
    if (NT_SUCCESS(status_dosname) && tmpDosName.Buffer != NULL) {
        ExFreePool(tmpDosName.Buffer);
    }
    
    return all_ok;
}

    USHORT sufixLen = FileName->Length - prefixNT.Length;

    // O sufixo começa 8 bytes depois do início do buffer original (pulando \??\)
    FileNameOk->Buffer = (PWCH)((PCHAR)FileName->Buffer + prefixNT.Length);
    FileNameOk->Length = sufixLen;

USHORT calcula_tamanho_final(PUNICODE_STRING FileName, USHORT* tamanhoDevName_cache) {
    USHORT retval = 0;

    // Caso 1: prefixo igual a \Device\HarddiskVolumeX
    if (RtlPrefixUnicodeString(&g_DevicePrefix, FileName, TRUE)) {
        retval = FileName->Length - g_DevicePrefix.Length + 4 + 10 * sizeof(WCHAR);
        if (g_Debug) DbgPrint("[SENTINELA] tamanho final com caso1: %d\n", retval);
    }
    // Caso 1.5: prefixo comeca com \Device mas ao eh g_DevicePrefix
    else if (RtlPrefixUnicodeString(&prefixDevice, FileName, TRUE))
    {
        *tamanhoDevName_cache = tamanho_deviceName(FileName);
        retval = FileName->Length - *tamanhoDevName_cache + 4 + 10 * sizeof(WCHAR);
        if (g_Debug) DbgPrint("[SENTINELA] tamanho final com caso15: %d\n", retval);
    }
    // 3. Caso: \SystemRoot
    else if (RtlPrefixUnicodeString(&prefixSysRoot, FileName, TRUE))
    {
        retval = FileName->Length - prefixSysRoot.Length + winRoot.Length + 10 * sizeof(WCHAR);
        if (g_Debug) DbgPrint("[SENTINELA] tamanho final com caso3: %d\n", retval);
    }

    return retval;
}

BOOLEAN converte_nome_arquivo(PUNICODE_STRING FileName, PUNICODE_STRING FileNameOk) {
    if (g_Debug) DbgPrint("[SENTINELA] converte_nome_arquivo: Iniciando com arquivo: %wZ\n", FileName);

    // 2. Caso: \??\C:\...
    if (RtlPrefixUnicodeString(&prefixNT, FileName, TRUE) && caso2(FileName, FileNameOk)) {
        // retorna FALSE sinalizando que nao foi alocado um novo buffer
        return FALSE;
    }

    USHORT tamanhoDevName_cache = 0;
    USHORT tamanho_necessario = calcula_tamanho_final(FileName, (USHORT*)&tamanhoDevName_cache);
    // se tamanho_necessario == 0 nao foi possivel calcular pq a entrada nao segue um padrao conhecido
    // nesse caso copia a entrada e ja era
    if (tamanho_necessario == 0) {
        DbgPrint("[SENTINELA] converte_nome_arquivo: Năo foi possível estimar o tamanho da string do nome, repassando o nome integralmente....");

        // fazendo o buffer de saida ser a copia apenas da estrutura da entrada sem copiar dados
        copia_unicode_data(FileName, FileNameOk);
        // retorna FALSE sinalizando que nao foi necessario alocar buffer
        return FALSE;
    }

    // alocando o buffer para o nome de arquivo convertido.
    PVOID tempBuffer = Malloc(tamanho_necessario, 'Temp');
    if (!tempBuffer) {
        // se nao conseguiu alocar o buffer faz a copia da entrada na saida e avisa o erro
        DbgPrint("[SENTINELA] converte_nome_arquivo: Năo foi possível alocar o buffer para o tamanho da string convertida: %d\n", tamanho_necessario);

        // fazendo o buffer de saida ser a copia apenas da estrutura da entrada sem copiar dados
        copia_unicode_data(FileName, FileNameOk);
        // retorna FALSE sinalizando que nao foi necessario alocar buffer
        return FALSE;

    }

    if (g_Debug) DbgPrint("[SENTINELA] converte_nome_arquivo: Alocado o buffer para string convertida: %d bytes\n", tamanho_necessario);

    // configurando FileNameOk 
    FileNameOk->Buffer = (PWCH)tempBuffer;
    FileNameOk->MaximumLength = tamanho_necessario;
    RtlZeroMemory(FileNameOk->Buffer, FileNameOk->MaximumLength);
    FileNameOk->Length = 0;

    // 1. Caso
    // se alterou o cache eh caso1 ou caso1.5 mas agora eh tudo uma coisa so
    // caso em que o caminho comeca com "\Device\"
    if (RtlPrefixUnicodeString(&prefixDevice, FileName, TRUE) && caso15(FileName, FileNameOk, tamanhoDevName_cache)) {
        DbgPrint("[SENTINELA] Resolvido com caso1 (unificado): %wZ\n", FileNameOk);
        return TRUE;
    }
    // 3. Caso: \SystemRoot
    else if (RtlPrefixUnicodeString(&prefixSysRoot, FileName, TRUE) && caso3(FileName, FileNameOk)) {
        DbgPrint("[SENTINELA] Resolvido com caso3: %wZ\n", FileNameOk);
        return TRUE;
    }
    else {
        DbgPrint("[SENTINELA] Resolvido com copia_nome_entrada: %wZ\n", FileNameOk);
        // se veio em um formato diferente ou se alguma funcao falhou em resolver passa integralmente
        // mas desaloca o buffer antes
        ExFreePoolWithTag(FileNameOk->Buffer, 'Temp');
        copia_unicode_data(FileName, FileNameOk);
        return FALSE;
    }
}
